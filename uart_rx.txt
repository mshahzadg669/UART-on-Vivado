`timescale 1ns / 1ps


module uart_rx (
    input  wire       clk,
    input  wire       rst,
    input  wire       rx_tick,   
    input  wire       rx_line,
    output reg  [7:0] rx_data,
    output reg        rx_done 
);

    reg [3:0] tick_count;
    reg [2:0] bit_count;
    reg [7:0] shift_reg;
    reg [1:0] State;

    localparam IDLE  = 2'b00;
    localparam START = 2'b01;
    localparam DATA  = 2'b10;
    localparam STOP  = 2'b11;

    always @(posedge clk or posedge rst) begin
        if (rst) begin
            State      <= IDLE;
            tick_count <= 0;
            bit_count  <= 0;
            shift_reg  <= 0;
            rx_data    <= 0;
            rx_done    <= 0;
        end else begin
            case (State)

                
                IDLE: begin
                    tick_count <= 0;   // BUG FIX 1: always clear here
                    bit_count  <= 0;
                    if (rx_line == 0) begin   
                        rx_done <= 0;         
                        State   <= START;
                    end
                end

                START: begin
                    if (rx_tick) begin
                        if (tick_count == 7) begin
                            
                            if (rx_line == 0) begin
                                tick_count <= 0;
                                bit_count  <= 0;
                                State      <= DATA;
                            end else begin
                                
                                State <= IDLE;
                            end
                        end else begin
                            tick_count <= tick_count + 1;
                        end
                    end
                end

                
                DATA: begin
                    if (rx_tick) begin
                        if (tick_count == 15) begin
                            tick_count <= 0;
                            
                            shift_reg  <= {rx_line, shift_reg[7:1]};
                            if (bit_count == 7) begin
                                State <= STOP;
                            end else begin
                                bit_count <= bit_count + 1;
                            end
                        end else begin
                            tick_count <= tick_count + 1;
                        end
                    end
                end

                
                STOP: begin
                    if (rx_tick) begin
                        if (tick_count == 15) begin
                            tick_count <= 0;
                            rx_data    <= shift_reg;
                            rx_done    <= 1;   
                            State      <= IDLE; 
                                               
                        end else begin
                            tick_count <= tick_count + 1;
                        end
                    end
                end

                default: State <= IDLE;
            endcase
        end
    end
endmodule
