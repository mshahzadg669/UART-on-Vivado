`timescale 1ns / 1ps


module BRG_tx( input wire clk,
               input wire rst,
               output reg tx_tick);

  reg [13:0] count;

  initial begin
    count<=0;
    tx_tick<=0;
  end
  
  always @(posedge clk or posedge rst)begin
	if(rst)begin
	count<=0;
	tx_tick<=0;
	end

    else if(count==10416)begin
      count<=0;
      tx_tick<=1;
    end
    else begin
      count<=count+1;
      tx_tick<=0;
    end
    
    
  end
  
  
endmodule
